使用 pass 语句，语法上来讲是冗余的，应该删除。

不符合规范的代码示例

    def __init__(self, log="", who="", date=0, files=[]):
        self.log = log
        self.files = files
        self.who = who
        self.date = date
        pass    # 不符合规范

    def lookup():
        pass    # 合规的; 路径不可以为空

合规的解决方案

    def __init__(self, log="", who="", date=0, files=[]):
        self.log = log
        self.files = files
        self.who = who
        self.date = date

    def lookup():
        pass
